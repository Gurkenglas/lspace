package main

import "fmt"

type Stage interface {
	Update()
}

type StageMenu struct {
	test int
}
type StageGame struct {
	test int
}

func (s *StageMenu) Update() {
	fmt.Printf("Updating Menu with value %d\n", s.test)
}

func (s *StageGame) Update() {
	fmt.Printf("Updating Stage with value %d\n", s.test)
}

func main() {
	DoSomething(&StageMenu{5})
	DoSomething(&StageGame{10})
}

func DoSomething(s Stage) {
	s.Update()
}
